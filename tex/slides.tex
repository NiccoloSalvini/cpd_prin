\section{\textbf{e-tourism}:Background and aim of the presentation}
\begin{frame}

\frametitle{Background: How much does tourism contribute to the economy?}
Tourism is one of the world’s fastest growing industries, contributing significantly to the economies of many countries.
\vspace{0.5cm}
\begin{itemize}
\item \textbf{Supply Side} $\rightarrow$ Tourism represent 4.5\% of the gross value added in the economy at EU level (Highest values: Croatia (11.3\%), Portugal (8.1\%), Spain (6.9\%) and Italy (6.2\%))

\vspace{0.2 cm}
\item \textbf{Demand Side}   $\rightarrow$  Per capita internal tourism consumption in the EU was estimated at EUR 2,794 per inhabitant 
\begin{itemize}
    \item [\ding{51}]  Luxembourg (EUR 5,831 per inhabitant), the Netherlands (EUR 5,255) and Austria (EUR 4,277) were on top.
    \item [\ding{51}] Italy in line with the EU average. 
\end{itemize}
\end{itemize}
\vspace{0.6 cm}
\tiny Source: Eurostat data on TSA (2023)
\end{frame}



\begin{frame}{\textbf{e-tourism}: Background and aim of the presentation}
\begin{itemize}
    \item The advent of the Internet has brought significant changes to the tourism sector, leading to the era of \textbf{digital tourism}. 
    \vspace{0.5cm}
    \item Main advantages of e-tourism: 

    \begin{itemize}
        \item [\ding{51}]  {\scriptsize \textbf{Enhanced Accessibility and Cost Efficiency} $\rightarrow$ The Internet provides extensive support for information, booking, and sales services, reaching a large number of tourists at relatively low costs.}
        \item [\ding{51}] {  \scriptsize \textbf{Improved Data Collection and Variety} $\rightarrow$ Technological advancements have significantly simplified the collection and storage of data, leading to a substantial increase in the variety of information available. }
    \end{itemize}
    \end{itemize}
    \vspace{0.3cm}
In order to monitor and manage tourism in view of e-tourism, granular and accurate spatio-temporal data is needed (Antolini et al., 2024)  $\rightarrow$ the use of unconventional data sources and big data analytics provide new opportunities for research (Arbia et al. 2024).  

\end{frame}

\begin{frame}{\textbf{e-tourism}: The Lack of Granular Price Data for Hotels}
\begin{itemize}
    \item Detailed price data on accommodation services across different geographical areas is \textbf{scarce}.
    \item Existing industry reports (e.g., \href{https://www.contenuti.icribis.com/osservatorio/2023/settore-alberghiero}{IX Report on the Italian Hospitality Industry}) provide insights on supply but lack precise price information.
    \item Platforms like \href{https://www.airdna.co/vacation-rental-data/app/it}{AirDNA} offer data primarily for short-term rentals (e.g., Airbnb, Vrbo), leaving gaps in hotel pricing analysis.
\end{itemize}
\end{frame}

\begin{frame}{\textbf{Aims of the Study}}
\begin{itemize}
    \item \textbf{Building a structured database} on hotel prices across Italian provinces to analyze spatial price differentials.
    \item Developing \textbf{spatial price indices} (SPI) to measure intra-national disparities, following the methodology used by Eurostat for international price comparisons.
    \item Providing \textbf{detailed and transparent information} to consumers on which cities and provinces have higher or lower accommodation costs.
    \item Offering a data-driven basis for \textbf{policy decisions} on tourism management, regional competitiveness, and pricing strategies.
\end{itemize}
\end{frame}

\section{Web Scraping: Architecture for Fast Insights }

\begin{frame}{Web Scraping: Architecture for Fast Insights}
\begin{itemize}
    \item \textbf{A Scalable and Efficient Approach}
    \begin{itemize}
        \item The DataLab @ Unitus has developed an automated system to collect and analyze hotel price data from online sources.
        \item The architecture is designed to ensure continuous data extraction, processing, and integration with analytical tools.
    \end{itemize}
    
    \item \textbf{Key Components:}
    \begin{itemize}
        \item A structured pipeline for web scraping, data storage, and statistical analysis.
        \item Dedicated modules for price monitoring, visualization, and reporting.
        \item Scalable infrastructure to handle large volumes of real-time data.
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Methodology and Infrastructure}
    \begin{itemize}
        \item The mechanisms encompasses:
        \begin{itemize}
            \item \textbf{API (Application Programming Interface) Reverse Engineering}
            Direct data extraction from structured API endpoints when available, ensuring efficiency and reliability.
            \item \textbf{Selenium-Based Automation}
            Employed for dynamic websites lacking API endpoints, allowing for automated interaction and data retrieval.
            \item \textbf{Distributed Scraping with Selenium Grid}
            Implements parallelized scraping across multiple machines, enhancing scalability and reducing execution time.
            \item \textbf{Traffic Routing via GlueTun}
            Ensures IP anonymization and circumvents access restrictions through proxy-based traffic routing.
        \end{itemize}
        \item The system is designed for adaptability, enabling rapid modifications in response to website structural changes or access restrictions.
    \end{itemize}
\end{frame}

\section{Data: web-scraped data from the CozyCozy platform }

\begin{frame}{CozyCozy: A Meta-Search Platform for Accommodation}
\begin{itemize}
    \item \textbf{What is CozyCozy?}
    \begin{itemize}
        \item CozyCozy is a \textbf{meta-search engine} for accommodation, aggregating listings from multiple booking platforms, including \textbf{Booking.com, Airbnb, Expedia, Trip.com}, and others.
        \item The platform enables users to compare prices and availability across different providers, offering a unified interface for accommodation search.
    \end{itemize}

    \item \textbf{Relevance for this Study}
    \begin{itemize}
        \item CozyCozy consolidates a diverse range of accommodation types, from hotels to vacation rentals, allowing for a comprehensive analysis of spatial price differences.
        \item The centralized nature of the platform facilitates efficient web scraping, ensuring broad market coverage and minimizing biases associated with single-platform data sources.
    \end{itemize}
\end{itemize}
\end{frame}


\begin{frame}{Data Description}

Some of the fields:

\begin{itemize}
    \item  {  \footnotesize \textbf{Hotel Name}: Name of the hotel %(\texttt{title}, \texttt{subTitle})
    }
    \item  {  \footnotesize \textbf{Geographic Location}: Coordinates }
    \item  {  \footnotesize \textbf{Price per night}: The recorded booking price %(\texttt{highlightedResults\_0\_totalPrice\_value}, \texttt{highlightedResults\_0\_totalPrice\_currencyCode})
    }
    \item {  \footnotesize  \textbf{Number of rooms and guests}: Number of rooms and guests for the booking %(\texttt{highlightedResults\_0\_bedRoomCount})
    }
    \item  {  \footnotesize \textbf{Rating Score}: User rating score and number of reviews %(\texttt{ratingScore}, \texttt{ratingCount})
    }
    \item {  \footnotesize \textbf{Rating Count}: Number of user who rated the the offer}
    \item {  \footnotesize \textbf{Services offered}: full-board, half-board, bed and brekfast, only bed }
    \item {  \footnotesize \textbf{Type of room: } single, double, ....}

    \item { \footnotesize \textbf{other offers: } vrbo.com 79, aibnb.com 65}

    \item {\footnotesize \textbf{Hotel details: } Amenities in the hotel}
    
\end{itemize}

\end{frame}

\begin{frame}{Data Collection: Frequency and Scope}
\begin{itemize}
    \item \textbf{Frequency}: Data is collected \textbf{daily at 12:00} through an automated web scraping pipeline.  
    \item \textbf{Coverage}: The dataset encompasses the entire Italian territory at the \textbf{provincial level}.  
    \item \textbf{Processing Time}: Each full scraping cycle takes approximately \textbf{3 hours} to complete.  

\end{itemize}
\end{frame}


\begin{frame}{Data Processing and Quality Control}
\begin{itemize}
    \item \textbf{Spatial Deduplication:}  
    \begin{itemize}
        \item DBSCAN clustering (50m radius) to merge duplicate listings.
        \item Fuzzy matching of hotel names to resolve inconsistencies.
        \item Price reconciliation for identical offers from different platforms.
    \end{itemize}

    \item \textbf{Outlier Removal:}  
    \begin{itemize}
        \item Elimination of extreme prices following industry standards and common sense (Choi et al., 2022).
        \item Category-specific thresholds for premium accommodations.
    \end{itemize}

    \item \textbf{Final Dataset:}  
    \begin{itemize}
        \item \textbf{3958} unique hotels across \textbf{1285} cities.
        \item Follows best practices in big data hedonic pricing (Li et al., 2023; Kim et al., 2021).
    \end{itemize}
\end{itemize}
\end{frame}

\section{SPI (Spatial Price Index)}


\begin{frame}{Hedonic CPD Method for Spatial Price Indices}
\begin{equation}
\ln P_{ij} = \sum_{j=1}^J \beta_j R_j + \sum_{i = 1}^N \alpha D_i +  \sum_{m=1}^{M} \delta_m R_m + u_{ij}
\end{equation}

% \sum_{k=1}^{K} \gamma_k L_k +

\begin{itemize}
    \item $P_{ij}$: Price of hotel $i$ in province $j$.
    \item $R_j$: Province dummies (regional effects).
    \item $D_i$: Hotel attributes.
    %\item $L_k$: \textbf{Location attributes} (distance to city center, attractions, local competition, socioeconomic factors).
    \item $R_m$: \textbf{Reputation attributes} (customer ratings, review count as trust indicators).
\end{itemize}

\begin{equation}
SPI_j = \exp(\beta_j)
\end{equation}

\textbf{Justification:}
\begin{itemize}
    %\item Location ($L_k$) affects accessibility, desirability, and competition (Choi et al., 2022).
    \item Reputation ($R_m$) influences consumer trust and willingness to pay (Kim et al., 2021).
    \item Hedonic CPD controls for quality differences, estimating pure spatial price differentials.
\end{itemize}

\end{frame}
